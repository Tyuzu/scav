const (
	CacheDir           = "./cache/media"
	CacheMaxAge        = 72 * time.Hour
	ClientTimeout      = 15 * time.Second
	MaxPixelsAllowed   = 4096 * 4096       // 16M pixels
	MaxImageBytes      = 200 * 1024 * 1024 // 200 MB read cap
	StreamThresholdPx  = 8 * 1024 * 1024   // >8M px -> stream
	ProbeLimit         = 64 * 1024         // bytes to probe for DecodeConfig
	MaxConcurrentFetch = 8
	MaxConcurrentOps   = 6
)

var (
	httpClient = &http.Client{Timeout: ClientTimeout}

	DomainAllowlist = map[string]bool{} // empty means allow unless blocked
	DomainBlocklist = map[string]bool{
		"malicious.example": true,
	}

	privatePrefixes = []string{
		"localhost", "127.", "10.", "192.168.", "172.",
	}

	fetchSem = make(chan struct{}, MaxConcurrentFetch)
	opSem    = make(chan struct{}, MaxConcurrentOps)
)

// ProxyHandler handles media proxy requests.
// route path expected: /proxy/:url  (router provides encoded URL as param)
func ProxyHandler(w http.ResponseWriter, r *http.Request, ps httprouter.Params) {
	raw := ps.ByName("url")
	target, err := normalizeTarget(raw)
	if err != nil {
		http.Error(w, "invalid url", http.StatusBadRequest)
		return
	}

	u, err := url.Parse(target)
	if err != nil {
		http.Error(w, "invalid url", http.StatusBadRequest)
		return
	}

	if !isAllowedHost(u.Hostname()) {
		http.Error(w, "blocked host", http.StatusForbidden)
		return
	}

	// ensure cache dir
	if err := os.MkdirAll(CacheDir, 0755); err != nil {
		http.Error(w, "server error", http.StatusInternalServerError)
		return
	}

	// cache key includes query (so transform params are part of key)
	cacheKey := target + "?" + r.URL.RawQuery
	cachePath := filepath.Join(CacheDir, hashURL(cacheKey))

	// serve cached if fresh
	if fi, err := os.Stat(cachePath); err == nil && time.Since(fi.ModTime()) < CacheMaxAge {
		http.ServeFile(w, r, cachePath)
		return
	}

	// parse client params
	wParam, _ := strconv.Atoi(r.URL.Query().Get("w"))
	hParam, _ := strconv.Atoi(r.URL.Query().Get("h"))
	qParam, _ := strconv.Atoi(r.URL.Query().Get("q"))
	if qParam <= 0 {
		qParam = 80
	}
	format := strings.ToLower(r.URL.Query().Get("format"))
	if format != "webp" && format != "avif" {
		format = "jpeg"
	}

	// perform a single upstream fetch
	resp, err := fetchRemote(httpClient, target)
	if err != nil {
		http.Error(w, "fetch failed", http.StatusBadGateway)
		return
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		http.Error(w, "upstream error", http.StatusBadGateway)
		return
	}

	contentType := resp.Header.Get("Content-Type")

	// non-image: stream to cache and serve
	if !strings.HasPrefix(contentType, "image/") {
		if err := streamToCache(resp.Body, cachePath); err != nil {
			http.Error(w, "stream fail", http.StatusBadGateway)
			return
		}
		if ct := resp.Header.Get("Content-Type"); ct != "" {
			w.Header().Set("Content-Type", ct)
		} else {
			w.Header().Set("Content-Type", "application/octet-stream")
		}
		w.Header().Set("Cache-Control", "public, max-age=86400")
		http.ServeFile(w, r, cachePath)
		return
	}

	// image: probe first by reading ProbeLimit bytes
	headBuf, err := io.ReadAll(io.LimitReader(resp.Body, ProbeLimit))
	if err != nil && err != io.EOF {
		http.Error(w, "probe fail", http.StatusBadGateway)
		return
	}

	// attempt to decode config from probe buffer
	cfg, _, err := image.DecodeConfig(bytes.NewReader(headBuf))
	if err != nil {
		// can't probe reliably -> stream the remainder (include probe bytes)
		if err := streamToCache(io.MultiReader(bytes.NewReader(headBuf), resp.Body), cachePath); err != nil {
			http.Error(w, "stream fail", http.StatusBadGateway)
			return
		}
		w.Header().Set("Content-Type", contentType)
		w.Header().Set("Cache-Control", "public, max-age=86400")
		http.ServeFile(w, r, cachePath)
		return
	}

	// check pixel limits
	totalPx := int64(cfg.Width) * int64(cfg.Height)
	if totalPx > StreamThresholdPx || totalPx > MaxPixelsAllowed {
		// too large -> stream original
		if err := streamToCache(io.MultiReader(bytes.NewReader(headBuf), resp.Body), cachePath); err != nil {
			http.Error(w, "stream fail", http.StatusBadGateway)
			return
		}
		w.Header().Set("Content-Type", contentType)
		w.Header().Set("Cache-Control", "public, max-age=86400")
		http.ServeFile(w, r, cachePath)
		return
	}

	// safe to fully read into memory (still capped)
	fullData, err := io.ReadAll(io.MultiReader(bytes.NewReader(headBuf), resp.Body))
	if err != nil {
		http.Error(w, "read fail", http.StatusBadGateway)
		return
	}
	if int64(len(fullData)) > MaxImageBytes {
		// too large to decode safely -> write what we have and serve (best-effort)
		if err := os.WriteFile(cachePath, fullData, 0644); err != nil {
			http.Error(w, "oversize", http.StatusBadGateway)
			return
		}
		w.Header().Set("Content-Type", contentType)
		w.Header().Set("Cache-Control", "public, max-age=86400")
		http.ServeFile(w, r, cachePath)
		return
	}

	img, _, err := image.Decode(bytes.NewReader(fullData))
	if err != nil {
		http.Error(w, "decode error", http.StatusBadGateway)
		return
	}

	// resize if requested
	if wParam > 0 || hParam > 0 {
		img = imaging.Resize(img, wParam, hParam, imaging.Lanczos)
	}

	// encode optimized image
	outBytes, outCT, err := encodeImage(img, format, qParam)
	if err != nil {
		http.Error(w, "encode fail", http.StatusBadGateway)
		return
	}

	// save to cache atomically and serve
	if err := saveAtomically(cachePath, outBytes); err != nil {
		// if caching fails, still return bytes
		w.Header().Set("Content-Type", outCT)
		w.Header().Set("Cache-Control", "public, max-age=0")
		_, _ = w.Write(outBytes)
		return
	}

	w.Header().Set("Content-Type", outCT)
	w.Header().Set("Cache-Control", "public, max-age=86400")
	http.ServeFile(w, r, cachePath)
}

// ---------------- helpers ----------------

func fetchRemote(client *http.Client, target string) (*http.Response, error) {
	fetchSem <- struct{}{}
	defer func() { <-fetchSem }()

	req, err := http.NewRequest("GET", target, nil)
	if err != nil {
		return nil, err
	}
	req.Header.Set("User-Agent", "MediaProxy/1.0")
	req.Header.Set("Accept", "*/*")
	return client.Do(req)
}

func normalizeTarget(raw string) (string, error) {
	raw = strings.TrimPrefix(raw, "/")
	u, err := url.Parse(raw)
	if err != nil || (u.Scheme != "http" && u.Scheme != "https") {
		return "", errors.New("invalid url")
	}
	return raw, nil
}

func isPrivateHost(host string) bool {
	host = strings.ToLower(host)
	if host == "localhost" || host == "::1" {
		return true
	}
	for _, p := range privatePrefixes {
		if strings.HasPrefix(host, p) {
			return true
		}
	}
	return false
}

func isBlockedHost(host string) bool {
	if isPrivateHost(host) {
		return true
	}
	return DomainBlocklist[host]
}

func isAllowedHost(host string) bool {
	if len(DomainAllowlist) > 0 {
		return DomainAllowlist[host]
	}
	return !isBlockedHost(host)
}

func hashURL(u string) string {
	h := sha1.New()
	h.Write([]byte(u))
	return hex.EncodeToString(h.Sum(nil))
}

func streamToCache(r io.Reader, path string) error {
	tmp := path + ".tmp"
	f, err := os.Create(tmp)
	if err != nil {
		return err
	}
	_, err = io.Copy(f, r)
	f.Close()
	if err != nil {
		_ = os.Remove(tmp)
		return err
	}
	return os.Rename(tmp, path)
}

func saveAtomically(path string, data []byte) error {
	tmp := path + ".tmp"
	if err := os.WriteFile(tmp, data, 0644); err != nil {
		return err
	}
	return os.Rename(tmp, path)
}
